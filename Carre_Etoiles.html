<!-- Double triangles -->
for (var j = 0; j < n; j++) {
    // Triangle des +
    for (var i = 0; i < (n - j); i++) {
        resultat += "+";
    }
    // Triangle des *
    for (var k = 0; k <= j; k++) {
        resultat += "*";
