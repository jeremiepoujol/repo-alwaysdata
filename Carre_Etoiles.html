<!-- Triangle d'étoiles -->
<script>
for (var j = 0; j < saisie; j++) {
    // À chaque ligne, on ajoute une étoile de plus
    for (var i = 0; i <= j; i++) {
        resultat += "*";

    resultat += "<br/>";

<!-- Double triangles -->
<script>
for (var j = 0; j < saisie; j++) {
    // Triangle des +
    for (var i = 0; i < (saisie - j); i++) {
        resultat += "+";
    }
    // Triangle des *
    for (var k = 0; k <= j; k++) {
        resultat += "*";
    }
    resultat += "<br/>";
}
</script>
